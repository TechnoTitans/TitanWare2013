005/**
006 * The purpose of this program is to compare the robot's ability to go in a

010public class GyroAllign {
011    //GearTrain driveForward;
012    TitanGyro gyroAllignVar; 
013    public double LEFT_MOTOR_SPEED_DEFAULT  = .3;
014    public double RIGHT_MOTOR_SPEED_DEFAULT = .3; 
015    double leftMotorS = .5;
016    double rightMotorS = .5;
017    
018    public double rightSpeed;
019    public double leftSpeed;
020    
021    public GyroAllign(TitanGyro passed) {
022        //riveForward = new GearTrain();
023        gyroAllignVar = passed;
024    }
025    
026    /**
027     * drives with no gyro assistance at speed .3 for both motors
028     */
029    public void noGyroForward () {
030        double leftMotorS = .3;
031        double rightMotorS = .3; 
032        //driveForward.setMotorSpeed(leftMotorS,rightMotorS);
033    }
034    /**
035     * Drives forward using gyro assistance
036     * threshold of 2 degrees
037     */
038    public void withGyroForward () {
039
040
041        double INCREASE_FACTOR = 1.25;
042        double angleLimit = 2; //lower threshold 
043        
044        //go forward while adjusting with gyro
045            //driveForward.setMotorSpeed(LEFT_MOTOR_SPEED_DEFAULT,RIGHT_MOTOR_SPEED_DEFAULT);
046            
047            double gyroAngle = gyroAllignVar.angle();
048            if (gyroAngle > angleLimit) {
049                //rightMotorS = rightMotorS * INCREASE_FACTOR; 
050                //driveForward.setMotorSpeed(LEFT_MOTOR_SPEED_DEFAULT,rightMotorS);  
051                rightSpeed = rightMotorS;
052                leftSpeed = LEFT_MOTOR_SPEED_DEFAULT;
053                //Timer.delay(0.2);
054                
055            }
056            else if (gyroAngle < -angleLimit) {
057                //leftMotorS = leftMotorS * INCREASE_FACTOR;
058                //driveForward.setMotorSpeed(leftMotorS, RIGHT_MOTOR_SPEED_DEFAULT);
059                leftSpeed = leftMotorS;
060                rightSpeed = RIGHT_MOTOR_SPEED_DEFAULT;
061                //Timer.delay(0.2);
062            }
063    }
064    /**
065     * gets speed of right motor
066     * @return speed of right motor
067     */
068    public double getRightSpeed () {
069        return rightSpeed;
070    }
071    /**
072     * gets speed of left motor
073     * @return speed of left motor
074     */
075    public double getLeftSpeed () {
076        return leftSpeed;
077    }
078    /**
079     * resets the gyro
080     */
081    public void gyroReset () {
082        gyroAllignVar.reset();
083        Timer.delay(.5);    
084    }
085}
