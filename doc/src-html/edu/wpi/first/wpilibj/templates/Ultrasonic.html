005package edu.wpi.first.wpilibj.templates;
006//import edu.wpi.first.wpilibj.Ultrasonic;
007import edu.wpi.first.wpilibj.AnalogChannel;
008/**
009 * @author Rohan Doshi
010 * Converts Voltage of ultrasonic to distance
011 */
012
013public class Ultrasonic {
014    public static final int ANALOG_CHANNEL = 3; // ? Correct
015    
016    AnalogChannel u;
017    public double ultrasonicVoltage;
018    public double distanceInches;
019    public double distanceMilimeters;
020    public double distanceMeters;
021    public double distanceFeet;
022    /* this class converts the analog input of our ultrasonic sensor
023     * into a readeable distance in inches and milimeters
024     */
025    
026    /**
027     * gets ultrasonic voltage and limits it
028     */
029    public Ultrasonic(){
030        u = new AnalogChannel(ANALOG_CHANNEL);
031    }
032    public double ultrasonicInput () {
033       ultrasonicVoltage = u.getAverageVoltage();
034       
035       if (ultrasonicVoltage <= 0) {
036           ultrasonicVoltage = 0;
037       } 
038       else if (ultrasonicVoltage >= 5) {
039           ultrasonicVoltage = 5;
040       } 
041       
042       //System.out.println(ultrasonicVoltage);
043       return ultrasonicVoltage;
044    }
045    /**
046     * Gets distance in Inches
047     * @return distanceInches
048     */
049    
050    public double ultrasonicInches () {
051        ultrasonicInput();
052        distanceInches = ultrasonicVoltage/0.009765;
053        //DriverStation.logData("DISTANCE IN INCHES", distanceInches);
054        return distanceInches;
055    }
056    /**
057     * Gives distance in Feet
058     * @return distanceFeet
059     */
060    public double ultrasonicFeet () {
061        ultrasonicInches();
062        distanceFeet = distanceInches/12;
063        return distanceFeet;
064    }
065    /**
066     * Gets distance in millimeters
067     * Divide by 1000 to get meters
068     * @return distanceMilimeters
069     */
070    public double ultrasonicMilimeters () {
071        ultrasonicInches();
072        distanceMilimeters = distanceInches*25.4;
073        //DriverStation.logData("DISTANCE IN MILIMETERS", distanceMilimeters);
074        return distanceMilimeters;
075    }
076    /**
077     * Gives distance in Meters
078     * @return distanceMeters
079     */
080    public double ultrasonicMeters () {
081        ultrasonicMilimeters();
082        distanceMeters = distanceMilimeters/1000;
083        return distanceMeters;
084    }
085
086}
